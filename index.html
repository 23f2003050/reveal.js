Scenario Analysis (Python)


import numpy as np

def cagr(begin, end, years):
    return (end / begin) ** (1/years) - 1

revenue = np.array([3.15, 3.42])  # $B
eps = np.array([1.19, 1.32])      # $

rev_growth = (revenue[-1]/revenue[0]) - 1
eps_growth = (eps[-1]/eps[0]) - 1
three_year_cagr = cagr(2.10, 3.42, 3)

print(f"QoQ Revenue Growth: {rev_growth:.2%}")
print(f"QoQ EPS Growth: {eps_growth:.2%}")
print(f"3Y CAGR (Revenue): {three_year_cagr:.2%}")
